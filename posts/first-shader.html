vec3 circle(vec2 uv, float rr, float cc, float ss) {
    uv*=mat2(cc,ss,-ss,cc);
    if (rr<0.) uv.y=-uv.y;
    rr=abs(rr);
    float r = length(uv)-rr;
    float pix=fwidth(r);
	float c = smoothstep(0.,pix,abs(r));
    float l = smoothstep(0.,pix,abs(uv.x)+step(uv.y,0.)+step(rr,uv.y));
   	return vec3(c,c*l,c*l);
}
